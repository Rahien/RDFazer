1 Introduction

RDFazer is a chrome browser plugin for tagging a website with RDFa concepts. It only tags the static page and does not store the content anywhere. It was concieved for taking an existing (static) webpage and tagging it with concepts from some taxonomy (e.g. the EC ESCO taxonomy). This page can then be stored elsewhere.

This code was written based on the nice input from:

the tutorial provided by google
e.g. https://developer.chrome.com/extensions/getstarted
stackoverflow
e.g. http://stackoverflow.com/questions/11684454/getting-the-source-html-of-the-current-page-from-chrome-extension

2 Manifest file

Chrome browser extensions require a json manifest file, dealing with permissions and setup of the extension. This is the file used for taggy: 

{

This will send a message to the injected script on the web page to initiate the user interface. After that, it will exchange messages with the injected script and the back-end.

function ajax(ajaxData,hasParams){
    ajaxData.url=ajaxData.url+(hasParams?"":"?")+"format=json";
    ajaxData.dataType="json";
    if(ajaxData.type!="GET"){
        ajaxData.data=JSON.stringify(ajaxData.data);
        ajaxData.contentType= "application/json; charset=utf-8";
        ajaxData.processData=false;
    }

    $.ajax(ajaxData);       
}

function handleServerError(request,status,error,stage){
    var message = "An error occurred during " + stage+".\n"+
        "The server responded with status code '"+status+"' and the error message was:\n'"+error+"'";
    taggyMessage(message,"bad-message");
}

function taggyMessage(message,classes){
    var body=document.getElementsByTagName("body");
    var messageDiv=document.createElement("div");
    $(messageDiv).addClass(classes+" taggyMessage").text(message);
    setTimeout(function(){
        $(messageDiv).addClass(classes+" taggyMessage-close")       
        setTimeout(function(){$(messageDiv).remove()},3000);
    },1500);
    $(body[0]).append(messageDiv);
    setTimeout(function(){
        $(messageDiv).addClass("taggyMessage-open");
    },0);
}

    readPanel:function(panel){
        lastValueItem = panel.preferredTerm[0].lastValueItem;
        var resultConcept = {
            type: panel.type,
            preferredTerm: lastValueItem.value,
            uri:lastValueItem.valueItem.uri
        };

        if(resultConcept.uri == ""){
            delete resultConcept.uri;
        }

        for(var name in panel.inputs){
            var inputDiv = panel.inputs[name];
            resultConcept[name]=inputDiv.inputType.readInput(inputDiv.input);
        }

        return resultConcept;
    },

    fillPanel:function(panel,values){
        for(var name in panel.inputs){
            var inputNode = panel.inputs[name];
            inputNode.inputType.fillInput(inputNode.input,values[name]);
        }
    },

    resetPanel:function(panel){
        for(var name in panel.inputs){
            var inputNode = panel.inputs[name];
            inputNode.inputType.fillInput(inputNode.input,null);
        }   
    },

    /**
     ,* Creates the input fields required by the given template
     ,*/
    buildInputFields:function(container,resultFormat){
        var inputs = {};
        for(var name in resultFormat){
            var inputType = resultFormat[name];
            var inputNode = inputTypes.buildInput(inputType,name);
            inputs[name] = inputNode;
            $(container).append(inputNode);
        }
        return inputs;
    },

    handleConceptSearchResults:function(request,handler,data,type){
        var elements = $.map( data, function( item ) {
            var valueItem= {
                label: item.preferredTerm,
                value: item.preferredTerm,
                valueItem: item
            };
            return valueItem;
        });
        var filteredElements = [];
        for(var i=0, element; element=elements[i]; i++){
            if(element.valueItem.type == type){
                filteredElements.push(element);
            }
        }

        elements=filteredElements;
        if(request.term){
            var useInputAsNew = {label:"<<new value: "+request.term+">>",
                                 value : request.term,
                                 valueItem : {preferredTerm:request.term,
                                              isNewConcept:true}};
            elements.push(useInputAsNew);
        }
        handler(elements);  
    }
};

var inputTypes = {
    string : {
        name : "string",
        buildSpecificInput : function(name){
            var temp = document.createElement("div");
            temp.innerHTML = "<input class='taggy-text' name='"+name+"'>";
            return temp.children[0];
        },
        readInput : function(input){
            return $(input).val();
        },
        fillInput : function(input,value){
            $(input).val(value?value:"");
        }
    },
    array : {
        name : "array",
        buildSpecificInput : function(name){
            var temp = document.createElement("div");
            temp.innerHTML = "<input class='taggy-array' name='"+name+"'>";
            return temp.children[0];
        },
        readInput : function(input){
            var stringvalue= $(input).val();
            var array =  stringvalue.split(",");
            for (var i=0, element; element = array[i]; i++){
                array[i] = element.trim();
            }
            return array;
        },
        fillInput : function(input,value){
            $(input).val(value?value.join(","):"");
        }
    },
    longText : {
        name : "longText",
        buildSpecificInput : function(name){
            var temp = document.createElement("div");
            temp.innerHTML = "<textarea class='taggy-long-text' type='textarea' name='"+name+"'></textarea>";
            return temp.children[0];
        },
        readInput : function(input){
            return $(input).val();
        },
        fillInput : function(input,value){
            $(input).val(value?value:"");
        }
    },
    uri : {
        name : "uri",
        buildSpecificInput : function(name){
            var temp = document.createElement("div");
            temp.innerHTML = "<input class='taggy-text' disabled='true' name='"+name+"'>";
            return temp.children[0];
        },
        readInput : function(input){
            return $(input).val();
        },
        fillInput : function(input,value){
            $(input).val(value?value:"");
        }
    },
    buildInput : function(inputType,name,args){
        var placeHolder=document.createElement("div");
        placeHolder.innerHTML="<div class = 'input'><span class='label'>"+name+"</span></div>";
        var node = placeHolder.children[0];
        var specific = inputType.buildSpecificInput(name,args);
        $(node).append(specific);
        node.inputType=inputType;
        node.input= specific;
        return node;
    }
};

5.2 Backend api

The following calls interact with the backend api to fetch and store information in the backend. The backend api can be included on the contentscript side as scripts there are allowed to do cross-origin requests. 

var lot3backend = {
    ToolsBackendURL:"http://localhost/lot3-tools/",  

    //* activates the input as an autocomplete search box for concepts, returning only concepts of a type
    setupSearch: function(input,type,panel,success,error,select){
        var self=this;
        var autoComplete = $(input).autocomplete({
            appendTo: input.parentNode,
            source: function( request, response ) {
                ajax({
                    type:"GET",
                    url: self.ToolsBackendURL+"concepts/search",
                    data: {
                        term: request.term
                    },
                    success: $.proxy(success,this,type,request,response),
                    error: error
                });
            },
            delay:500,
            minLength: 3,
            select: function(event,ui){
                input.lastValueItem = ui.item;
                select(event,ui);
            }
        });
        return autoComplete;        
    },

    //* stores the given concept in the backend
    saveConcept:function(concept,success,error){
        if(concept.uri){
            // existing concept
            var self = this;
            ajax({
                type:"PUT",
                url: self.ToolsBackendURL+"concept",
                data: concept,
                success: success,
                error: error
            });
        }else{
            // new concept
            var self = this;
            ajax({
                type:"POST",
                url: self.ToolsBackendURL+"concepts",
                data: concept,
                success: success,
                error: error
            });

        }
    }
}
